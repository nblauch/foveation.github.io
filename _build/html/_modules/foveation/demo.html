def get_image_as_batch(path='shark.png', device='cuda'):
    img = Image.open(path).convert('RGB')
    batch = TF.to_tensor(img).unsqueeze(0).to(device)
    batch = TF.normalize(batch, mean=[0.485, 0.456, 0.406], std=[0.229, 0.224, 0.225])
    return batch
