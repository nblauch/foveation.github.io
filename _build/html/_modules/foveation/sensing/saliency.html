@add_to_all(__all__)
def sample_from_logdensity(log_density, count=1, tmp_scale=1.0, ior_width=None):
    """Sample coordinates from a log density map.
    
    Args:
        log_density: Tensor of shape (batch_size, height, width) containing log density values
        count: Number of samples per batch item (default: 1)
        tmp_scale: Temperature scaling factor for sampling (default: 1.0)
        ior_width: If not None, applies inhibition of return by setting probabilities 
                  within this radius of previous samples to zero
    
    Returns:
        Tuple of (x_coords, y_coords, log_probs) where:
            x_coords: Tensor of shape (batch_size, count) or (batch_size,) if count=1
            y_coords: Same shape as x_coords
            log_probs: Tensor of shape (batch_size, height, width) containing final log probabilities
    """
    try:
        batch_size, height, width = log_density.shape
    except:
        batch_size, _, height, width = log_density.shape

    # Flatten spatial dimensions
    log_probs = log_density.reshape(batch_size, -1)
    
    # Handle invalid values
    valid_mask = torch.isfinite(log_probs)
    log_probs = log_probs.masked_fill(~valid_mask, float('-inf'))
    
    values = []
    for i in range(count):
        # apply softmax w/ temperature to get probabilities
        probs = torch.exp(log_probs / tmp_scale) / torch.sum(torch.exp(log_probs / tmp_scale), dim=1, keepdim=True)

        # sample according to probabilities
        dist = torch.distributions.Categorical(probs=probs)
        fixation = dist.sample((1,))
        values.append(fixation)

        if ior_width is not None:
            # Apply inhibition of return
            fix_xs = fixation % width
            fix_ys = fixation // width
            x = torch.arange(width)
            y = torch.arange(height)
            Y, X = torch.meshgrid(x, y)
            X = X.reshape(1, -1).repeat(batch_size, 1).transpose(0, 1).to(log_probs.device)
            Y = Y.reshape(1, -1).repeat(batch_size, 1).transpose(0, 1).to(log_probs.device)
            circ_inds = torch.sqrt((X - fix_xs)**2 + (Y - fix_ys)**2) < ior_width
            log_probs[circ_inds.transpose(0, 1)] = -float('inf')

    # Combine all samples
    samples = torch.cat(values, dim=0).transpose(0, 1)
    
    # Convert to x,y coordinates
    x_coords = samples % width
    y_coords = samples // width
    
    # Reshape log probabilities back to spatial form
    log_probs = log_probs.reshape(batch_size, height, width)

    if count == 1:
        return x_coords.squeeze(-1), y_coords.squeeze(-1), log_probs

    return x_coords, y_coords, log_probs

def get_center_bias(res_in=256, device='cuda',
    path='/n/home12/nblauch/git/saccadenet-private/submodules/DeepGaze/centerbias_mit1003.npy'
):
    centerbias_template = np.load(path)
    # rescale to match image size
    centerbias = zoom(centerbias_template, (res_in/1024, res_in/1024), order=0, mode='nearest')
    # renormalize log density
    centerbias -= logsumexp(centerbias)
    centerbias_tensor = torch.tensor([centerbias]).to(device)
    return centerbias_tensor

@add_to_all(__all__)
def sample_from_logits_pc(logits, coords, size=1, tmp_scale=None, bounding_coords=None, deterministic=False):
    """ Sample from log probabilities (robust to many bins and small probabilities).
        +-inf and nan will be interpreted as zero probability
        
        Args:
            logits: Tensor of shape (batch_size, num_points)
            coords: Tensor of shape (batch_size, num_points, 2) containing x,y coordinates
            size: Number of samples per batch item
            tmp_scale: Scale parameter for exponential distribution
            bounding_coords: Bounding coordinates for fixations
            deterministic: Whether to sample deterministically as the argmax over logits
            
        Returns:
            Tuple of (sampled_indices, updated_logits) where:
                sampled_indices: Tensor of shape (batch_size, size) containing sampled point indices
                updated_logits: Tensor of shape (batch_size, num_points) with IOR applied
    """
    batch_size, num_points = logits.shape
    if deterministic:
        assert tmp_scale is None or tmp_scale == 1
    if tmp_scale is None:
        tmp_scale = 1.0
    # Generate random exponential values
    # tmps = -torch.distributions.Exponential(1/tmp_scale).sample((batch_size, size)).to(logits.device)
    tmps = torch.ones((batch_size, size)).to(logits.device) * tmp_scale

    # Find valid (finite) indices and probabilities
    valid_mask = torch.isfinite(logits)
    if bounding_coords is not None:
        assert len(bounding_coords) == 2
        assert bounding_coords[0] < bounding_coords[1] and bounding_coords[0] >= -1 and bounding_coords[1] <= 1
        bound_mask = (coords > bounding_coords[0]) & (coords < bounding_coords[1])
        bound_mask = bound_mask.all(dim=2)
        valid_mask = valid_mask & bound_mask
    
    # Get valid log probabilities and normalize them
    valid_logits = logits.masked_fill(~valid_mask, float('-inf'))
    # logits_norm = valid_logits - torch.logsumexp(valid_logits, dim=1, keepdim=True)

    values = []
    for ii in range(size):
        # Convert to probabilities and scale by temperature
        probs = F.softmax(valid_logits / tmps[:,ii].unsqueeze(1), dim=1)
        
        # Sample indices according to probabilities
        if deterministic:
            fixation = torch.argmax(probs, dim=1).unsqueeze(0)
        else:
            dist = torch.distributions.Categorical(probs=probs)
            fixation = dist.sample((1,))
        values.append(fixation)

    values = torch.cat(values, dim=0).transpose(0, 1)
        
    return values, valid_logits
