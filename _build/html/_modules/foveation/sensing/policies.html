    def get_random_crop(self, height, width, scale, ratio):
        """
        Generate a random crop with specified scale and aspect ratio.
        
        Args:
            height (int): Image height.
            width (int): Image width.
            scale (float or tuple): Scale factor(s) for crop area.
            ratio (float or tuple): Aspect ratio(s) for the crop.
            
        Returns:
            tuple:
                - list: Normalized fixation center [y, x]
                - list: Fixation size [height, width]
        """
        area = height * width
        log_ratio = np.log(ratio)
        for _ in range(10):
            if hasattr(scale, '__len__'):
                target_area = area * np.random.uniform(scale[0], scale[1])
            else:
                target_area = area * scale
            if hasattr(ratio, '__len__'):
                aspect_ratio = np.exp(np.random.uniform(log_ratio[0], log_ratio[1]))
            else:
                aspect_ratio = ratio
            w = int(round(np.sqrt(target_area * aspect_ratio)))
            h = int(round(np.sqrt(target_area / aspect_ratio)))
            if 0 < w <= width and 0 < h <= height:
                i = int(np.random.uniform(0, height - h + 1))
                j = int(np.random.uniform(0, width - w + 1))
                # calculate normalized center of fixation
                fixation = [(i+h/2)/height,(j+w/2)/width]
                fixation_size = [h,w]
                return fixation, fixation_size
        in_ratio = float(width) / float(height)
        if in_ratio < min(ratio):
            w = width
            h = int(round(w / min(ratio)))
        elif in_ratio > max(ratio):
            h = height
            w = int(round(h * max(ratio)))
        else:
            w = width
            h = height
        i = (height - h) // 2
        j = (width - w) // 2

        # calculate normalized center of fixation
        fixation = [(i+h/2)/height,(j+w/2)/width]
        fixation_size = [h,w]

        return fixation, fixation_size

    def get_random_nearcenter_fixation(self, height, width, scale, ratio, normalized_dist_from_center):
        """
        Generate a random fixation near the center with specified constraints.
        
        Args:
            height (int): Image height.
            width (int): Image width.
            scale (float or tuple): Scale factor(s) for crop area.
            ratio (float or tuple): Aspect ratio(s) for the crop.
            normalized_dist_from_center (float): Maximum normalized distance from center.
            
        Returns:
            tuple:
                - list: Normalized fixation center [y, x]
                - list: Fixation size [height, width]
        """
        area = height * width
        log_ratio = np.log(ratio)

        if hasattr(scale, '__len__'):
            target_area = area * np.random.uniform(scale[0], scale[1])
        else:
            target_area = area * scale
        if hasattr(ratio, '__len__'):
            aspect_ratio = np.exp(np.random.uniform(log_ratio[0], log_ratio[1]))
        else:
            aspect_ratio = ratio

        if isinstance(target_area, torch.Tensor):
            target_area = target_area.item()

        w = int(round(np.sqrt(target_area * aspect_ratio)))
        h = int(round(np.sqrt(target_area / aspect_ratio)))
        fixation_size = [h,w]

        min_frac = 0.5-normalized_dist_from_center
        max_frac = 0.5+normalized_dist_from_center
        fixation = [torch.tensor(np.random.uniform(min_frac, max_frac)), torch.tensor(np.random.uniform(min_frac, max_frac))]

        return fixation, fixation_size

    def sample_fixations(self, img_size, n=1, area_range=None, ratio=None, norm_dist_from_center=None):
        """
        Sample multiple fixations for batch processing.
        
        Args:
            img_size (tuple): Image size (height, width).
            n (int): Number of fixations to sample. Defaults to 1.
            area_range: Scale range for crop area. Defaults to None.
            ratio: Aspect ratio range. Defaults to None.
            norm_dist_from_center (float, optional): Maximum normalized distance from center. Defaults to None.
            
        Returns:
            tuple:
                - torch.Tensor: Fixation locations of shape (n, 2)
                - np.ndarray: Fixation sizes of shape (n, 2)
        """
        if area_range is None:
            area_range = self.crop_area_range if self.training else self.val_crop_size
        if ratio is None:
            ratio = self.crop_aspect_range if self.training else 1
        fixations = []
        fixation_sizes = []
        for _ in range(n):
            if norm_dist_from_center is not None:
                fixation, fixation_size = self.get_random_nearcenter_fixation(img_size[0], img_size[1], scale=area_range, ratio=ratio, 
                    normalized_dist_from_center=norm_dist_from_center,
                    )
            else:
                fixation, fixation_size = self.get_random_crop(img_size[0], img_size[1], scale=area_range, ratio=ratio)
            fixations.append(fixation)
            fixation_sizes.append(fixation_size)
        fixations = torch.tensor(fixations, dtype=self.dtype, device=self.device)
        fixation_sizes = torch.tensor(fixation_sizes)
        return fixations, fixation_sizes

    """
    This class implements a multi-random saccade policy for generating fixations in images. The policy randomly selects multiple fixation points within the image.

    Attributes:
        retinal_transform (RetinalTransform): The retinal transform object used to apply retinal transformations to the images.
        n_fixations (int): The number of fixations to generate.

        nonrandom_val (bool): whether to make validation fixations deterministic
        norm_dist_from_center (float): if not None, changes how fixations are sampled. rather than finding any valid crop, it takes a fixation within norm_dist_from_center fractional distance from the center of the image. 
    """

    def forward(self, x, n_fixations=None, fixations=None, fixation_size=None, area_range=None):
        """
        Forward pass for the MultiRandomSaccadePolicy.

        This method generates multiple random fixations for the input images and applies the retinal transform to each fixation.

        Args:
            x (torch.Tensor): The input images of shape (n, c, h, w), where n is the batch size, c is the number of channels, h is the height, and w is the width.
            n_fixations (int, optional): The number of fixations to generate. Defaults to None, which uses the default number of fixations set in the policy.
            fixations (list of torch.Tensor, optional): A list of pre-defined fixations. Defaults to None, which generates random fixations.
            fixation_size (int, optional): The size of the fixation area. Defaults to None, which uses the default fixation size set in the policy.
            area_range (tuple, optional): The range of areas to sample from for the fixation size. Defaults to None.

        Returns:
            dict: Dictionary containing:
                - x_fixs (torch.Tensor): The transformed images.
                - fixations (torch.Tensor): The fixation coordinates.
                - fixation_sizes (torch.Tensor): The fixation sizes.
                - fix_deltas (torch.Tensor): The fixation deltas.
        """
        if n_fixations is None:
            n_fixations = self.n_fixations

        assert fixations is None or len(fixations) == n_fixations    

        fixation_sizes = []
        fixations_ = []
        for fix_i in range(n_fixations):
            fix_, fix_size = self.sample_fixations(x.shape[2:], n=x.shape[0], area_range=area_range, norm_dist_from_center=self.norm_dist_from_center)
            fixation_sizes.append(fix_size)
            if fix_i == 0 and self.nonrandom_first:
                fix_ = torch.tensor([0.5, 0.5]).to(dtype=fix_.dtype, device=fix_.device)
            fixations_.append(fix_)
        if fixations is None:
            if self.nonrandom_val and not self.training:
                fixations = [0.5, 0.5]*n_fixations
            else:
                fixations = fixations_
        else:
            assert len(fixations) == n_fixations
            for ii, fixation in enumerate(fixations):
                if len(fixation) == 2:
                    fixation = torch.ones(x.shape[0], 2)*fixation
                    fixations[ii] = fixation
                 
        x_fixs = []
        fix_deltas = []
        for ii, (fixation, fixation_size) in enumerate(zip(fixations, fixation_sizes)):
            x_fix = self.retinal_transform(x, fixation, fixation_size=fixation_size)  
            x_fixs.append(x_fix)
            if ii > 0:
                fix_deltas.append(fixation - fixations[ii-1])
            else:
                fix_deltas.append(torch.zeros(x.shape[0], 2, device=x.device))
        
        return {
            'x_fixs': torch.stack(x_fixs, dim=1), # (B, F, C, N)
            'fixations': torch.stack(fixations, dim=1), # (B, F, 2) 
            'fixation_sizes': torch.stack(fixation_sizes, dim=1), # (B, F, 2) 
            'fix_deltas': torch.stack(fix_deltas, dim=1), # (B, F, 2) 
            }


    
    def __repr__(self):
        return f"MultiRandomSaccadePolicy(retinal_transform={self.retinal_transform}, n_fixations={self.n_fixations})"

class DeepGazeISaccadePolicy(BaseSaccadePolicy):
    """
    Salience-based fixation policy that uses a pre-trained DeepGazeI model to generate fixations.
    """
    def __init__(self, retinal_transform, n_fixations=1, res_in=256, ior_width=None, tmp_scale=1, batch_size=None):
        super().__init__(retinal_transform, n_fixations)
        if not DEEPGAZE_ALLOWED:
            raise RuntimeError('please install deepgaze_pytorch')
        self.saliency_model = DeepGazeI(pretrained=True, res_in=res_in, normalize=False)
        self.center_bias = get_center_bias(res_in=res_in)
        self.ior_width = ior_width
        self.tmp_scale = tmp_scale
        self.res_in = res_in
        self.batch_size = batch_size # in case we want to use a reduced batch size, due to the model being large
        self.cached_saliency = None

    @torch.no_grad()
    def get_saliency(self, x):
        if self.retinal_transform.pre_transforms is not None and self.training:
            x = self.retinal_transform.pre_transforms(x)
        if self.batch_size is None:
            log_density_prediction = self.saliency_model(x, centerbias=self.center_bias.to(x.device))
        else:
            # split x into batches and concatenate predictions
            predictions = []
            for i in range(0, x.shape[0], self.batch_size):
                batch_x = x[i:np.minimum(i+self.batch_size,x.shape[0])]
                batch_pred = self.saliency_model(batch_x, centerbias=self.center_bias.to(x.device))
                predictions.append(batch_pred)
            log_density_prediction = torch.cat(predictions, dim=0)
        return log_density_prediction

    def load_cached_saliency(self, cache_fn=None, cached_saliency=None):
        if cached_saliency is None:
            self.cached_saliency = torch.load(cache_fn)
        else:
            self.cached_saliency = cached_saliency

    def forward(self, x, n_fixations=None, fixation_size=None, area_range=None, fixations=None, cache_inds=None):
        if n_fixations is None:
            n_fixations = self.n_fixations

        assert fixations is None or len(fixations) == n_fixations    

        fixation_sizes = []
        for fix_i in range(n_fixations):
            _, fix_size = self.sample_fixations(x.shape[2:], n=x.shape[0], area_range=area_range)
            fixation_sizes.append(fix_size)

        if cache_inds is not None:
            assert self.cached_saliency is not None
            log_density_prediction = self.cached_saliency[cache_inds]
        else:
            log_density_prediction = self.get_saliency(x)

        fix_x, fix_y, log_density_post_ior = sample_from_logdensity(log_density_prediction, count=n_fixations, tmp_scale=self.tmp_scale, ior_width=self.ior_width)

        fixations = torch.stack([fix_y/self.res_in, fix_x/self.res_in], dim=2)
        fixations = [fixations[:,ii] for ii in range(n_fixations)]

        x_fixs = []
        fix_deltas = []
        for ii, (fixation, fixation_size) in enumerate(zip(fixations, fixation_sizes)):
            x_fix = self.retinal_transform(x, fixation, fixation_size=fixation_size)  
            x_fixs.append(x_fix)
            if ii > 0:
                fix_deltas.append(fixation - fixations[ii-1])
            else:
                fix_deltas.append(torch.zeros(x.shape[0], 2, device=x.device))

        return {
            'x_fixs': torch.stack(x_fixs, dim=1), # (B, F, C, N)
            'fixations': torch.stack(fixations, dim=1), # (B, F, 2) 
            'fixation_sizes': torch.stack(fixation_sizes, dim=1), # (B, F, 2) 
            'fix_deltas': torch.stack(fix_deltas, dim=1), # (B, F, 2) 
            }

class NetworkSaliencySaccadePolicy(BaseSaccadePolicy):
    """
    Salience-based fixation policy that uses intermediate activations from a specified network layer
    to generate a saliency map and sample fixations. The saliency map is updated after each fixation
    based on the network's processing of the transformed input.

    Attributes:
        retinal_transform (RetinalTransform): The retinal transform object
        network (nn.Module): The network to extract intermediate activations from
        hook_layer (nn.Module): the layer to extract intermedate activations from
        intermediate_coords (nn.Module): intermediate coordinates corresponding to hook_layer
        n_fixations (int): Number of fixations to generate
        ior_width (float, optional): Width for inhibition of return
        tmp_scale (float): Temperature scaling for sampling
    """
    def __init__(self, retinal_transform, network, hook_layer, intermediate_coords,
                 n_fixations=1, ior_width=None, tmp_scale=1, deterministic=False, norm_dim=1,
                 remove_leading_tokens=False,
                 ):
        super().__init__(retinal_transform, n_fixations)
        self.network = network
        self.hook_layer = hook_layer
        self.intermediate_coords = intermediate_coords
        self.ior_width = ior_width
        self.tmp_scale = tmp_scale
        self.deterministic = deterministic
        self.norm_dim = norm_dim
        self.remove_leading_tokens = remove_leading_tokens
        
        # Register hook to capture intermediate activations
        self.activation = None
        self._register_activation_hook()

    def _register_activation_hook(self):
        """
        Register a forward hook on the target layer to capture its activations.
        """
        def hook_fn(module, input, output):
            self.activation = output

        self.hook_layer.register_forward_hook(hook_fn)

        # set up sampler for inhibition map
        self.intermed_sampler = GridSampler(None, None, None, coords=self.intermediate_coords)
        self.intermed_sampling_grid = self.intermed_sampler.coords.cartesian.unsqueeze(0).unsqueeze(0)

    @torch.no_grad()
    def _compute_saliency_map(self, activation, inhibition_map=None):
        """
        Convert intermediate activations into a saliency map.
        
        Args:
            activation (torch.Tensor): Intermediate activations [batch, channels, height, width]
            
        Returns:
            torch.Tensor: Saliency map [batch, height, width]
        """
        # take l2-norm across channels (softmax w/ temperature is done in the sampling function)
        saliency = F.relu(activation).norm(p=2, dim=self.norm_dim)

        # remove leading tokens, e.g. cls and buffer tokens, if they are present
        if self.remove_leading_tokens:
            n_tokens = saliency.shape[1] - inhibition_map.shape[1]
            saliency = saliency[:,n_tokens:]

        # multiply by inhibition map if provided
        if inhibition_map is not None:
            saliency = saliency * (1-inhibition_map)
                
        # locally pool the saliency map
        # TO-DO
        
        return saliency

    def _update_inhibition_map(self, x, inhibition_map, fixation):
        """
        Update the inhibition map based on the fixation location.
        """
        if inhibition_map is None:
            inhibition_map = torch.zeros((x.shape[0], x.shape[2], x.shape[3])).to(x.device)

        # add a local hot spot of inhibition on the fixation 
        x_grid = torch.arange(0, x.shape[2])
        y_grid = torch.arange(0, x.shape[3])
        X, Y = torch.meshgrid(x_grid, y_grid)
        X = X.unsqueeze(0).repeat(x.shape[0], 1, 1).to(inhibition_map.device)
        Y = Y.unsqueeze(0).repeat(x.shape[0], 1, 1).to(inhibition_map.device)
        dists = torch.sqrt((X-x.shape[2]*fixation[:,0].unsqueeze(-1).unsqueeze(-1))**2 + (Y-x.shape[3]*fixation[:,1].unsqueeze(-1).unsqueeze(-1))**2)
        local_inhibition = dists < self.ior_width
        inhibition_map[local_inhibition] = 1
        return inhibition_map

    def forward(self, x, n_fixations=None, fixation_size=None, area_range=None, fixations=None):
        """
        Generate fixations based on network activations, updating saliency after each fixation.
        
        Args:
            x (torch.Tensor): Input images [batch, channels, height, width]
            n_fixations (int, optional): Number of fixations to generate
            fixation_size (int, optional): Size of fixation area
            area_range (tuple, optional): Range for sampling fixation area
            fixations (list, optional): Pre-defined fixations
            
        Returns:
            dict: Contains outputs, fixations, sizes, network outputs and other metadata
        """
        if n_fixations is None:
            n_fixations = self.n_fixations

        image_size = x.shape[2:]

        assert fixations is None or len(fixations) == n_fixations

        # Lists to store results
        x_fixs = []
        fix_deltas = []
        fixation_sizes = []
        fixations_list = []
        saliency_maps = []
        net_outputs = []
        global_grids = []
        all_fix_inds = []
        
        # first fixation in center 
        _, fix_size = self.sample_fixations(
            x.shape[2:], n=x.shape[0], area_range=area_range)
        fixation_sizes.append(fix_size)
        fixation = torch.tensor([0.5, 0.5]).unsqueeze(0).repeat(x.shape[0], 1).to(x.device)
        fixations_list.append(fixation.clone())
        prev_fixation = fixation

        # add a local hot spot of inhibition on the fixation 
        if self.ior_width is not None:
            inhibition_map = self._update_inhibition_map(x, None, fixation)
        else:
            inhibition_map = None

        # Process each fixation sequentially
        for fix_i in range(n_fixations):

            # Get network activations and compute saliency
            x_fix = self.retinal_transform(x, prev_fixation, fixation_size=fix_size)
            x_fixs.append(x_fix)
            net_output = self.network(x_fix)  # Forward pass to trigger hook
            # get inhibition map in fixated space
            inhibition_fixation = self.intermed_sampler(inhibition_map.unsqueeze(1), fix_loc=prev_fixation, fixation_size=fix_size).squeeze(1)
            
            fix_saliency = self._compute_saliency_map(self.activation, inhibition_map=inhibition_fixation)

            # determine global coordinates of saliency map
            global_grid = transform_sampling_grid(
                self.intermed_sampling_grid.repeat(x.shape[0], 1, 1, 1), 
                prev_fixation, fix_size, image_size,
                )
            global_grids.append(global_grid.squeeze(1))
                
            # Store network output
            net_outputs.append(net_output)

            # Sample next fixation point from saliency map
            fix_inds, fix_saliency = sample_from_logits_pc(
                fix_saliency,  # Add small epsilon to avoid log(0)
                global_grid.squeeze(1),
                size=1,
                tmp_scale=self.tmp_scale,
                bounding_coords = [-0.8, 0.8],
                deterministic=self.deterministic,
            )
            # fixation = torch.cat([fix_y, fix_x], dim=1)
            fixation = torch.gather(global_grid.squeeze(1), 1, fix_inds.unsqueeze(-1).expand(-1, -1, 2))
            # swap x and y
            fixation = torch.cat([fixation[:,:,1], fixation[:,:,0]], dim=1)
            # fixation = torch.cat([fixation[:,:,0], fixation[:,:,1]], dim=1)
            # convert from [-1, 1] to [0,1]
            fixation = (fixation + 1) / 2
            fixations_list.append(fixation.clone())
            all_fix_inds.append(fix_inds)

            saliency_maps.append(fix_saliency)
            
            # Compute fixation delta
            if prev_fixation is not None:
                fix_deltas.append(fixation - prev_fixation)
            else:
                fix_deltas.append(torch.zeros(x.shape[0], 2, device=x.device))
            prev_fixation = fixation

            # Generate next fixation size
            _, fix_size = self.sample_fixations(
                x.shape[2:], n=x.shape[0], area_range=area_range)
            fixation_sizes.append(fix_size)

            # update inhibition map
            if self.ior_width is not None:
                inhibition_map = self._update_inhibition_map(x, inhibition_map, fixation)

        return {
            'x_fixs': torch.stack(x_fixs, dim=1), # (B, F, C, N)
            'fixations': torch.stack(fixations, dim=1), # (B, F, 2) 
            'fixation_sizes': torch.stack(fixation_sizes, dim=1), # (B, F, 2) 
            'fix_deltas': torch.stack(fix_deltas, dim=1), # (B, F, 2) 
            'saliency_maps': torch.stack(saliency_maps, dim=1),
            'net_outputs': torch.stack(net_outputs, dim=1),
            'global_grids': torch.stack(global_grids, dim=1),
            'fix_inds': torch.stack(all_fix_inds, dim=1),
        }

    def __repr__(self):
        return f"NetworkSaliencySaccadePolicy(network={type(self.network).__name__}, layer_idx={self.layer_idx}, layer_type={self.layer_type}, n_fixations={self.n_fixations})"

                - x_fixs (torch.Tensor): The transformed image.
                - fixations (torch.Tensor): The fixation coordinates.
                - fixation_sizes (torch.Tensor): The fixation sizes.
                - fixation_deltas (torch.Tensor): The fixation deltas.
        """
        assert f1 is None and area_range is None and n_fixations is None
        x_f1 = self.retinal_transform(x, f1)
        x_f1 = x_f1.unsqueeze(0)
        return {
            'x_fixs': x_f1,
            'fixations': 0.5*torch.ones((x_f1.shape[0], 1, 2)),
            'fixation_sizes': self.fixation_size*torch.ones((x_f1.shape[0], 1, 2)),
            'fixation_deltas': torch.zeros(x_f1.shape[0], 1, 2),
        }


    
    def __repr__(self):
        return f"NoSaccadePolicy(retinal_transform={self.retinal_transform})"
